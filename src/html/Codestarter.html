function calculateBMI(height, weight) {
    const bmi = weight / (height * height);
    return bmi.toFixed(2);
}

function checkHealthStatus(bmi) {
    if (bmi < 18.5) {
        return "Underweight";
    } else if (bmi < 25) {
        return "Healthy";
    } else {
        return "Overweight";
    }
}
